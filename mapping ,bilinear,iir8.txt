clc;
close all;
fs=1000;
fn=fs/2;
fc=300;
n=5;
[z,p,k]=butter(n,fc/fn);
b=k*poly(z);
a=poly(p);
[h,om]=freqz(b,a,512,fs);
subplot(2,1,1);
plot(om,20*log10(abs(h)));
xlabel('normalized frequency');
ylabel('gain in db');
title('magnitude response');
subplot(2,1,2);
plot(om,angle(h));
xlabel('normalized frequency');
ylabel('phase in radian');
title('phase response');
