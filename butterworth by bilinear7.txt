clc;
close all;
alphas=30;
alphap=0.5;
fpass=1000;
fstop=1500;
fsam=5000;
wp=2*fpass/fsam;
ws=2*fstop/fsam;
[n,wn]=buttord(wp,ws,alphap,alphas);
[b,a]=butter(n,wn);
[h,w]=freqz(b,a);
subplot(2,1,1);
plot(w/pi,20*log10(abs(h)));
xlabel('normalized frequency');
ylabel('Gain in db');
title('Magnitude Response');
subplot(2,1,2);
plot(w/pi,angle(h));
xlabel('Normalized frequency');
ylabel('Phase in Radian');
title('Phase Response');
